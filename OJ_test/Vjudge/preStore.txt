#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n, target;
    cin >> n >> target;
    vector<int> v(n);
    for (int i = 0; i < n; i++)
    {
        cin >> v[i];
    }
    sort(v.begin(), v.end());
    int count = 0;
    int res = 0;
    for (int i = n - 1; i >= 0; i--)
    {
        count += v[i];
        if (count >= target)
        {
            res++;
            break;
        }
        res++;
    }
    cout << res;
}

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin >> n;
    vector<int> v(n);
    for (int i = 0; i < n; i++)
    {
        cin >> v[i];
    }
    int count = 0;
    for (int i = 0; i < n - 1; i++)
    {
        for (int j = 0; j < n - i - 1; j++)
        {
            if (v[j] > v[j + 1])
            {
                swap(v[j], v[j + 1]);
                count++;
            }
        }
    }
    cout << count;
}

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin >> n;
    vector<long long> v(n);
    for (int i = 0; i < n; i++)
    {
        cin >> v[i];
    }
    vector<long long> minus(n - 1);
    for (int i = 0; i < n - 1; i++)
    {
        minus[i] = abs(v[i + 1] - v[i]);
    }
    sort(minus.begin(), minus.end());
    for (int i = 1; i <= n - 1; i++)
    {
        if (i != minus[i - 1])
        {
            cout << "Not jolly";
            return 0;
        }
    }
    cout << "Jolly";
}

#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;
double distance(vector<int> *a, int size)
{
    double sum = 0;
    for (int i = 0; i < size - 1; i++)
    {
        sum += sqrt(pow(a[i][0] - a[i + 1][0], 2) + pow(a[i][1] - a[i + 1][1], 2) + pow(a[i][2] - a[i + 1][2], 2));
    }
    return sum;
}
int main()
{
    int n;
    cin >> n;
    vector<int> *a = new vector<int>[n];
    for (int i = 0; i < n; i++)
    {
        int x, y, z;
        cin >> x >> y >> z;
        a[i].push_back(x);
        a[i].push_back(y);
        a[i].push_back(z);
    }
    sort(a, a + n, [](vector<int> &a, vector<int> &b)
         { return a[2] < b[2]; });
    printf("%.3f", distance(a, n));

    delete[] a;
}

            return a.i > b.i; });

#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

bool cmp(string a, string b)
{
    return a + b > b + a;
}

int main()
{
    int n;
    cin >> n;
    vector<string> s(n);
    for (int i = 0; i < n; i++)
        cin >> s[i];
    sort(s.begin(), s.end(), cmp);

    string t = "";
    for (int i = 0; i < n; i++)
    {
        t += s[i];
    }
    while(t.size() > 1)
    {
        if(t[0] == '0')
        {
            t.erase(0, 1);
        }
        else
            break;
    }
    cout << t;
}
