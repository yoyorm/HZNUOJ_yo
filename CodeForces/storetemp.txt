#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while (T--)
    {
        int n;
        cin >> n;
        vector<vector<int>> v(n + 2, vector<int>(n + 2, 0));
        int count = 1;
        for (int i = 1; i <= n; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                v[i][j] = count;
                count++;
            }
        }
        int sumMAX = 0;
        for (int i = 1; i <= n; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                int sum = 0;
                sum = v[i][j] + v[i - 1][j] + v[i + 1][j] + v[i][j - 1] + v[i][j + 1];
                if (sum > sumMAX)
                    sumMAX = sum;
            }
        }
        cout << sumMAX<<endl;
    }
}

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while (T--)
    {
        int n;
        cin >> n;
        vector<int> v(n, 0);
        for (int i = 0; i < n; i++)
        {
            cin >> v[i];
        }
        sort(v.begin(), v.end());
        int max = 0;
        for (int i = 0; i < n; i++)
        {
            if (v[i] > max)
            {
                max = v[i];
            }
        }

        int count = 0;
        if (n == max)
            for (int i = 0; i < n; i++)
            {
                if (v[i] != 0)
                {
                    count = i;
                    break;
                }
            }
        else
        {
            int time = n - 1;
            for (int i = n-1; i > 0; i--)
            {
                while (v[i] > 1 && time > 0)
                {
                    v[i]--;
                    time--;
                }
            }
            for (int i = n-1; i > 0; i--)
            {
                if (time > 0)
                {
                    v[i]--;
                    time--;
                }
                else
                    break;
            }
            sort(v.begin(), v.end());
            for (int i = 0; i < n; i++)
            {
                if (v[i] != 0)
                {
                    count = i;
                    break;
                }
            }
        }
        cout << n - count << endl;
    }
}
